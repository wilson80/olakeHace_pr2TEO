DROP DATABASE IF EXISTS proyectoHelloWorld;

CREATE DATABASE proyectoHelloWorld;
USE proyectoHelloWorld;
-- Crear base de datos

CREATE TABLE equipos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL,
    pais VARCHAR(100),
    fundacion YEAR,
    imagen_path VARCHAR(255), -- path de la imagen del equipo
    partidas_jugadas INT DEFAULT 0,
    partidas_ganadas INT DEFAULT 0
);

CREATE TABLE jugadores (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL,
    equipo_id INT,
    edad INT NOT NULL,
    altura DECIMAL(5,2),
    peso DECIMAL(5,2),
    nacionalidad VARCHAR(100),
    fecha_insercion DATE DEFAULT CURRENT_DATE,
    total_anotaciones INT DEFAULT 0,
    partidos_ganados INT DEFAULT 0,
    partidos_jugados INT DEFAULT 0,
    imagen_path VARCHAR(255), -- pathh de la imagen del jugador
    FOREIGN KEY (equipo_id) REFERENCES equipos(id)
);

CREATE TABLE encuentros (
    id INT AUTO_INCREMENT PRIMARY KEY,
    equipo_local_id INT,
    equipo_visitante_id INT,
    goles_local INT DEFAULT 0,
    goles_visitante INT DEFAULT 0,
    fecha DATE,
    lugar VARCHAR(255),
    edicion VARCHAR(100),
    resultado VARCHAR(50), -- 'Equipo Local', 'Equipo Visitante', 'Empate'
    FOREIGN KEY (equipo_local_id) REFERENCES equipos(id),
    FOREIGN KEY (equipo_visitante_id) REFERENCES equipos(id)
);
